## TypeScript 示例

```typescript
interface User {
  id: number;
  name: string;
  email: string;
}

const users: User[] = [
  { id: 1, name: "Alice", email: "alice@example.com" }
];

export default users;
```

## JavaScript 示例

```javascript
function hello(name) {
  console.log(`Hello, ${name}!`);
}

hello("World");
```

## Python 示例

```python
def greet(name: str) -> str:
    return f"Hello, {name}!"

print(greet("World"))
```
